---
title: "disable"
description: "Disable passkeys (WebAuthn) authentication"
---

## Import

```ts
import { disable } from '@rhinestone/sdk/actions/passkeys'
```

## Usage

```ts
const transaction = await account.prepareTransaction({
  chain,
  calls: [disable()],
})
```
